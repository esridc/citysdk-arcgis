Map

Comparing Census counties with Maryland priority funding areas. Priority Funding Areas are existing communities and places where local governments want State investment to support future growth.

Run Demo

Example Code

var map = L.map('demomap').setView([38.81403111409755, -76.849365234375], 7);
L.esri.basemapLayer("Gray").addTo(map);

var request = {
  "state": "MD",
  "sublevel": true,
  "level": "state"
};

var callback = function(response) {
  var myLayer = L.geoJson().addTo(map);
  myLayer.addData(response);
};

census.GEORequest(request, callback);

var parks = new L.esri.FeatureLayer({
 url: "http://mdpgis.mdp.state.md.us/arcgis/rest/services/PlanningCadastre/Priority_Funding_Areas/MapServer/0",
 style: function () {
    return { color: "#70ca49", weight: 2 };
  }
}).addTo(map);
